    <script>
        const tg = window.Telegram.WebApp;
        tg.expand(); // Розгортаємо вікно на максимум

        // Отримуємо дані гравця з Telegram
        const user = tg.initDataUnsafe?.user;
        if (user) {
            document.getElementById('user-name').innerText = user.username ? '@' + user.username : user.first_name;
            document.getElementById('user-id').innerText = 'ID: ' + user.id;
            if (user.first_name) {
                document.getElementById('user-avatar').innerText = user.first_name.charAt(0).toUpperCase();
            }
        }

        let currentGame = "";

        function openGame(gameName, icon) {
            currentGame = gameName;
            document.getElementById('modal-game-title').innerText = icon + " " + gameName;
            document.getElementById('game-modal').classList.add('active');
        }

        function closeModal() {
            document.getElementById('game-modal').classList.remove('active');
        }

        // Логіка швидкої симуляції гри прямо в Web App для відправки результату бекенду
        function playRound() {
            const betInput = document.getElementById('bet-amount');
            const betValue = parseInt(betInput.value);

            if (isNaN(betValue) || betValue <= 0) {
                alert("Будь ласка, введіть правильну суму ставки!");
                return;
            }

            closeModal();

            // Випадковий результат (виграв чи програв)
            const isWin = Math.random() > 0.52; // Шанс виграшу близький до реального казино
            let change = 0;
            let textResult = "";

            if (isWin) {
                const multiplier = (1.2 + Math.random() * 2).toFixed(2);
                const winAmount = Math.round(betValue * multiplier);
                change = winAmount - betValue;
                textResult = `🎉 **Виграш!** Множник: **x${multiplier}**\n➕ Зараховано: **+${winAmount} зірок**`;
            } else {
                change = -betValue;
                textResult = `📉 **Програш!** Удача посміхнеться наступного разу.\n➖ Списано: **-${betValue} зірок**`;
            }

            // Пакет даних для нашого Python бота
            const gameData = {
                game: currentGame,
                bet: betValue,
                change: change,
                text: textResult
            };

            // Надсилаємо дані назад у бот і закриваємо Web App сторінку
            tg.sendData(JSON.stringify(gameData));
            tg.close();
        }

        function triggerAction(actionName) {
            tg.sendData(JSON.stringify({ action: actionName }));
            tg.close();
        }
    </script>
